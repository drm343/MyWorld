  77 static Yes_No can_move_horizon(Camera_Access access,
  78     Point_Type point) {
  79   Point_Type center = access->center;
  80 
  81   Point_Type from = access->start;
  82   Point_Type to = access->end;
  83 
  84   Map_Access map = access->map;
  85   Yes_No result = YES;
  86 
  87   if ((point.x > (center.x - 1))
  88       && ((point.x + 1)< (map->end.x - (center.x - 1)))) {
  89     access->horizon = CAMERA_MOVE;
  90   }
  91   else if (from.x <= 0) {
  92     access->horizon = CAMERA_FIX;
  93 
  94     if (point.x < 0) {
  95       result = NO;
  96     }
  97   }
  98   else if (to.x >= map->end.x) {
  99     access->horizon = CAMERA_FIX;
 100 
 101     if (point.x >= map->end.x) {
 102       result = NO;
 103     }
 104   }
 105   else {
 106     access->horizon = CAMERA_MOVE;

 145 static bool key_process(Camera_Access access, Character_Pool_Access from_pool,
 146     SDL_Event *event) {
 147   Status_Access Player = access->player;
 148   Status_Access npc = NULL;
 149   Style_Access dead = access->dead;
 150 
 151   Point_Type point = {
 152     Player->base->Real_Position.x,
 153     Player->base->Real_Position.y
 154   };
 155 
 156   Point_Type vector = {0, 0};
 157 
 158   if ((event->key).keysym.sym == SDLK_q) {
 159     return false;
 160   }
 161 
 162   switch ((event->key).keysym.sym) {
 163     case SDLK_UP:
 164       point.y -= 1;
 165       vector.y = -1;
 166       break;
 167     case SDLK_DOWN:
 168       point.y += 1;
 169       vector.y = 1;
 170       break;
 171     case SDLK_LEFT:
 172       point.x -= 1;
 173       vector.x = -1;
 174       break;
 175     case SDLK_RIGHT:
 176       point.x += 1;
 177       vector.x = 1;
 178       break;
 179     default:
 180       printf("other\n");
 181       break;
 182   }
 183 
 184   if (occupy_position_by_others(from_pool, point, &npc) == NO) {
 185     if ((vector.y != 0) && (can_move_vertical(access, point) == YES)) {
 186       switch (access->vertical) {
 187         case CAMERA_MOVE:
 188           access->start.y = access->start.y + vector.y;
 189           access->end.y = access->end.y + vector.y;
 190           break;
 191         default:
 192           break;
 193       }
 194 
 195       Player->base->Real_Position.x = Player->base->Real_Position.x + vector.x;
 196       Player->base->Real_Position.y = Player->base->Real_Position.y + vector.y;
 197     }
 198     else if ((vector.x != 0) && (can_move_horizon(access, point) == YES)) {
 199       switch (access->horizon) {
 200         case CAMERA_MOVE:
 201           access->start.x = access->start.x + vector.x;
 202           access->end.x = access->end.x + vector.x;
 203           break;
 204         default:
 205           break;
 206       }
 207 
 208       Player->base->Real_Position.x = Player->base->Real_Position.x + vector.x;
 209       Player->base->Real_Position.y = Player->base->Real_Position.y + vector.y;
 210     }
 211   }
 212   else {
 213     if (character.attack(Player, npc) == DEAD) {
 214       character.set_style(npc, dead);
 215     }
 216   }
 217   character_pool.reset_graph_position(from_pool, access->start.x, access->start.y);
 218   return true;
 219 }
