  27 /*

  28 ** In mpc the input type has three modes of 

  29 ** operation: String, File and Pipe.

  30 **

  31 ** String is easy. The whole contents are 

  32 ** loaded into a buffer and scanned through.

  33 ** The cursor can jump around at will making 

  34 ** backtracking easy.

  35 **

  36 ** The second is a File which is also somewhat

  37 ** easy. The contents are never loaded into 

  38 ** memory but backtracking can still be achieved

  39 ** by seeking in the file at different positions.

  40 **

  41 ** The final mode is Pipe. This is the difficult

  42 ** one. As we assume pipes cannot be seeked - and 

  43 ** only support a single character lookahead at 

  44 ** any point, when the input is marked for a 

  45 ** potential backtracking we start buffering any 

  46 ** input.

  47 **

  48 ** This means that if we are requested to seek

  49 ** back we can simply start reading from the

  50 ** buffer instead of the input.

  51 **

  52 ** Of course using `mpc_predictive` will disable

  53 ** backtracking and make LL(1) grammars easy

  54 ** to parse for all input methods.

  55 **

  56 */

 235 static void *mpc_malloc(mpc_input_t *i, size_t n) {
 236   size_t j;
 237   char *p;
 238   
 239   if (n > sizeof(mpc_mem_t)) { return malloc(n); }
 240   
 241   j = i->mem_index;
 242   do {
 243     if (!i->mem_full[i->mem_index]) {
 244       p = (void*)(i->mem + i->mem_index);
 245       i->mem_full[i->mem_index] = 1;
 246       i->mem_index = (i->mem_index+1) % MPC_INPUT_MEM_NUM;
 247       return p;
 248     }
 249     i->mem_index = (i->mem_index+1) % MPC_INPUT_MEM_NUM;
 250   } while (j != i->mem_index);
 251   
 252   return malloc(n);
 253 }

 299 static void mpc_input_mark(mpc_input_t *i) {
 300   
 301   if (i->backtrack < 1) { return; }
 302   
 303   i->marks_num++;
 304   
 305   if (i->marks_num > i->marks_slots) {
 306     i->marks_slots = i->marks_num + i->marks_num / 2;
 307     i->marks = realloc(i->marks, sizeof(mpc_state_t) * i->marks_slots);
 308     i->lasts = realloc(i->lasts, sizeof(char) * i->marks_slots);
 309   }
 310 
 311   i->marks[i->marks_num-1] = i->state;
 312   i->lasts[i->marks_num-1] = i->last;
 313   
 314   if (i->type == MPC_INPUT_PIPE && i->marks_num == 1) {
 315     i->buffer = calloc(1, 1);
 316   }
 317   
 318 }

 432 static int mpc_input_failure(mpc_input_t *i, char c) {
 433 
 434   switch (i->type) {
 435     case MPC_INPUT_STRING: { break; }
 436     case MPC_INPUT_FILE: fseek(i->file, -1, SEEK_CUR); { break; }
 437     case MPC_INPUT_PIPE: {
 438       
 439       if (!i->buffer) { ungetc(c, i->file); break; }
 440       
 441       if (i->buffer && mpc_input_buffer_in_range(i)) {
 442         break;
 443       } else {
 444         ungetc(c, i->file); 
 445       }
 446     }
 447     default: { break; }
 448   }
 449   return 0;
 450 }

 452 static int mpc_input_success(mpc_input_t *i, char c, char **o) {
 453   
 454   if (i->type == MPC_INPUT_PIPE
 455   &&  i->buffer && !mpc_input_buffer_in_range(i)) {
 456     i->buffer = realloc(i->buffer, strlen(i->buffer) + 2);
 457     i->buffer[strlen(i->buffer) + 1] = '\0';
 458     i->buffer[strlen(i->buffer) + 0] = c;
 459   }
 460   
 461   i->last = c;
 462   i->state.pos++;
 463   i->state.col++;
 464   
 465   if (c == '\n') {
 466     i->state.col = 0;
 467     i->state.row++;
 468   }
 469   
 470   if (o) {
 471     (*o) = mpc_malloc(i, 2);
 472     (*o)[0] = c;
 473     (*o)[1] = '\0';
 474   }
 475   
 476   return 1;
 477 }

 723 static mpc_err_t *mpc_err_or(mpc_input_t *i, mpc_err_t** x, int n) {
 724   
 725   int j, k, fst;
 726   mpc_err_t *e;
 727   
 728   fst = -1;
 729   for (j = 0; j < n; j++) {
 730     if (x[j] != NULL) { fst = j; }
 731   }
 732   
 733   if (fst == -1) { return NULL; }
 734   
 735   e = mpc_malloc(i, sizeof(mpc_err_t));
 736   e->state = mpc_state_invalid();
 737   e->expected_num = 0;
 738   e->expected = NULL;
 739   e->failure = NULL;
 740   e->filename = mpc_malloc(i, strlen(x[fst]->filename)+1);
 741   strcpy(e->filename, x[fst]->filename);
 742   
 743   for (j = 0; j < n; j++) {
 744     if (x[j] == NULL) { continue; }
 745     if (x[j]->state.pos > e->state.pos) { e->state = x[j]->state; }
 746   }
 747   
 748   for (j = 0; j < n; j++) {
 749     if (x[j] == NULL) { continue; }
 750     if (x[j]->state.pos < e->state.pos) { continue; }
 751     
 752     if (x[j]->failure) {
 753       e->failure = mpc_malloc(i, strlen(x[j]->failure)+1);
 754       strcpy(e->failure, x[j]->failure);
 755       break;
 756     }
 757     
 758     e->recieved = x[j]->recieved;
 759     
 760     for (k = 0; k < x[j]->expected_num; k++) {
 761       if (!mpc_err_contains_expected(i, e, x[j]->expected[k])) {
 762         mpc_err_add_expected(i, e, x[j]->expected[k]);
 763       }
 764     }
 765   }
 766   
 767   for (j = 0; j < n; j++) {
 768     if (x[j] == NULL) { continue; }
 769     mpc_err_delete_internal(i, x[j]);
 770   }
 771   
 772   return e;
 773 }

 776

 796     x->expected[0] = expect;
 797     return x;

 799   
 800   else if (x->expected_num > 1) {
 801     
 802     l += strlen(prefix);
 803     for (j = 0; j < x->expected_num-2; j++) {
 804       l += strlen(x->expected[j]) + strlen(", ");
 805     }
 806     l += strlen(x->expected[x->expected_num-2]);
 807     l += strlen(" or ");
 808     l += strlen(x->expected[x->expected_num-1]);
 809     
 810     expect = mpc_malloc(i, l + 1);
 811     
 812     strcpy(expect, prefix);
 813     for (j = 0; j < x->expected_num-2; j++) {
 814       strcat(expect, x->expected[j]); strcat(expect, ", ");

 816     strcat(expect, x->expected[x->expected_num-2]);

 818     strcat(expect, x->expected[x->expected_num-1]);

 821     
 822     x->expected_num = 1;
 823     x->expected = mpc_realloc(i, x->expected, sizeof(char*) * x->expected_num);
 824     x->expected[0] = expect;
 825     return x;
 826   }
 827   
 828   return NULL;
 829 }

1004 #define MPC_FAILURE(x) r->error = x; return 0
1005 #define MPC_PRIMITIVE(x) \

1009 static int mpc_parse_run(mpc_input_t *i, mpc_parser_t *p, mpc_result_t *r, mpc_err_t **e) {
1010   
1011   int j = 0, k = 0;
1012   mpc_result_t results_stk[MPC_PARSE_STACK_MIN];
1013   mpc_result_t *results;
1014   int results_slots = MPC_PARSE_STACK_MIN;
1015   
1016   switch (p->type) {
1017       
1018     /* Basic Parsers */
1019 
1020     case MPC_TYPE_ANY:     MPC_PRIMITIVE(mpc_input_any(i, (char**)&r->output));
1021     case MPC_TYPE_SINGLE:  MPC_PRIMITIVE(mpc_input_char(i, p->data.single.x, (char**)&r->output));
1022     case MPC_TYPE_RANGE:   MPC_PRIMITIVE(mpc_input_range(i, p->data.range.x, p->data.range.y, (char**)&r->output));
1023     case MPC_TYPE_ONEOF:   MPC_PRIMITIVE(mpc_input_oneof(i, p->data.string.x, (char**)&r->output));
1024     case MPC_TYPE_NONEOF:  MPC_PRIMITIVE(mpc_input_noneof(i, p->data.string.x, (char**)&r->output));
1025     case MPC_TYPE_SATISFY: MPC_PRIMITIVE(mpc_input_satisfy(i, p->data.satisfy.f, (char**)&r->output));
1026     case MPC_TYPE_STRING:  MPC_PRIMITIVE(mpc_input_string(i, p->data.string.x, (char**)&r->output));
1027     case MPC_TYPE_ANCHOR:  MPC_PRIMITIVE(mpc_input_anchor(i, p->data.anchor.f, (char**)&r->output));
1028     
1029     /* Other parsers */
1030     
1031     case MPC_TYPE_UNDEFINED: MPC_FAILURE(mpc_err_fail(i, "Parser Undefined!"));
1032     case MPC_TYPE_PASS:      MPC_SUCCESS(NULL);
1033     case MPC_TYPE_FAIL:      MPC_FAILURE(mpc_err_fail(i, p->data.fail.m));
1034     case MPC_TYPE_LIFT:      MPC_SUCCESS(p->data.lift.lf());
1035     case MPC_TYPE_LIFT_VAL:  MPC_SUCCESS(p->data.lift.x);
1036     case MPC_TYPE_STATE:     MPC_SUCCESS(mpc_input_state_copy(i));
1037     
1038     /* Application Parsers */
1039     
1040     case MPC_TYPE_APPLY:
1041       if (mpc_parse_run(i, p->data.apply.x, r, e)) {
1042         MPC_SUCCESS(mpc_parse_apply(i, p->data.apply.f, r->output));
1043       } else {
1044         MPC_FAILURE(r->output);
1045       }
1046     
1047     case MPC_TYPE_APPLY_TO:
1048       if (mpc_parse_run(i, p->data.apply_to.x, r, e)) {
1049         MPC_SUCCESS(mpc_parse_apply_to(i, p->data.apply_to.f, r->output, p->data.apply_to.d));
1050       } else {
1051         MPC_FAILURE(r->error);
1052       }
1053     
1054     case MPC_TYPE_EXPECT:
1055       mpc_input_suppress_enable(i);
1056       if (mpc_parse_run(i, p->data.expect.x, r, e)) {
1057         mpc_input_suppress_disable(i);
1058         MPC_SUCCESS(r->output);
1059       } else {
1060         mpc_input_suppress_disable(i);
1061         MPC_FAILURE(mpc_err_new(i, p->data.expect.m));
1062       }
1063     
1064     case MPC_TYPE_PREDICT:
1065       mpc_input_backtrack_disable(i);
1066       if (mpc_parse_run(i, p->data.predict.x, r, e)) {      
1067         mpc_input_backtrack_enable(i);
1068         MPC_SUCCESS(r->output);
1069       } else {
1070         mpc_input_backtrack_enable(i);
1071         MPC_FAILURE(r->error);
1072       }
1073     
1074     /* Optional Parsers */
1075     
1076     /* TODO: Update Not Error Message */
1077     
1078     case MPC_TYPE_NOT:
1079       mpc_input_mark(i);
1080       mpc_input_suppress_enable(i);
1081       if (mpc_parse_run(i, p->data.not.x, r, e)) {
1082         mpc_input_rewind(i);
1083         mpc_input_suppress_disable(i);
1084         mpc_parse_dtor(i, p->data.not.dx, r->output);
1085         MPC_FAILURE(mpc_err_new(i, "opposite"));
1086       } else {
1087         mpc_input_unmark(i);
1088         mpc_input_suppress_disable(i);
1089         MPC_SUCCESS(p->data.not.lf());
1090       }
1091     
1092     case MPC_TYPE_MAYBE:
1093       if (mpc_parse_run(i, p->data.not.x, r, e)) {
1094         MPC_SUCCESS(r->output);
1095       } else {
1096         *e = mpc_err_merge(i, *e, r->error);
1097         MPC_SUCCESS(p->data.not.lf());
1098       }
1099     
1100     /* Repeat Parsers */
1101     
1102     case MPC_TYPE_MANY:
1103       
1104       results = results_stk;
1105       
1106       while (mpc_parse_run(i, p->data.repeat.x, &results[j], e)) {
1107         j++;
1108         if (j == MPC_PARSE_STACK_MIN) {
1109           results_slots = j + j / 2;
1110           results = mpc_malloc(i, sizeof(mpc_result_t) * results_slots);
1111           memcpy(results, results_stk, sizeof(mpc_result_t) * MPC_PARSE_STACK_MIN);
1112         } else if (j >= results_slots) {
1113           results_slots = j + j / 2;
1114           results = mpc_realloc(i, results, sizeof(mpc_result_t) * results_slots);
1115         }
1116       }
1117       
1118       *e = mpc_err_merge(i, *e, results[j].error);
1119       MPC_SUCCESS(
1120         mpc_parse_fold(i, p->data.repeat.f, j, (mpc_val_t**)results);
1121         if (j >= MPC_PARSE_STACK_MIN) { mpc_free(i, results); });
1122     
1123     case MPC_TYPE_MANY1:
1124       
1125       results = results_stk;
1126       
1127       while (mpc_parse_run(i, p->data.repeat.x, &results[j], e)) {
1128         j++;
1129         if (j == MPC_PARSE_STACK_MIN) {
1130           results_slots = j + j / 2;
1131           results = mpc_malloc(i, sizeof(mpc_result_t) * results_slots);
1132           memcpy(results, results_stk, sizeof(mpc_result_t) * MPC_PARSE_STACK_MIN);
1133         } else if (j >= results_slots) {
1134           results_slots = j + j / 2;
1135           results = mpc_realloc(i, results, sizeof(mpc_result_t) * results_slots);
1136         }
1137       }
1138       
1139       if (j == 0) {
1140         MPC_FAILURE(
1141           mpc_err_many1(i, results[j].error);
1142           if (j >= MPC_PARSE_STACK_MIN) { mpc_free(i, results); });
1143       } else {
1144         *e = mpc_err_merge(i, *e, results[j].error);
1145         MPC_SUCCESS(
1146           mpc_parse_fold(i, p->data.repeat.f, j, (mpc_val_t**)results);
1147           if (j >= MPC_PARSE_STACK_MIN) { mpc_free(i, results); });
1148       }
1149     
1150     case MPC_TYPE_COUNT:
1151       
1152       results = p->data.repeat.n > MPC_PARSE_STACK_MIN
1153         ? mpc_malloc(i, sizeof(mpc_result_t) * p->data.repeat.n)
1154         : results_stk;
1155       
1156       while (mpc_parse_run(i, p->data.repeat.x, &results[j], e)) {
1157         j++;
1158         if (j == p->data.repeat.n) { break; }
1159       }
1160       
1161       if (j == p->data.repeat.n) {
1162         MPC_SUCCESS(
1163           mpc_parse_fold(i, p->data.repeat.f, j, (mpc_val_t**)results);
1164           if (p->data.repeat.n > MPC_PARSE_STACK_MIN) { mpc_free(i, results); });
1165       } else {
1166         for (k = 0; k < j; k++) {
1167           mpc_parse_dtor(i, p->data.repeat.dx, results[k].output);
1168         }
1169         MPC_FAILURE(
1170           mpc_err_count(i, results[j].error, p->data.repeat.n);
1171           if (p->data.repeat.n > MPC_PARSE_STACK_MIN) { mpc_free(i, results); });  
1172       }
1173       
1174     /* Combinatory Parsers */
1175     
1176     case MPC_TYPE_OR:
1177       
1178       if (p->data.or.n == 0) { MPC_SUCCESS(NULL); }
1179       
1180       results = p->data.or.n > MPC_PARSE_STACK_MIN
1181         ? mpc_malloc(i, sizeof(mpc_result_t) * p->data.or.n)
1182         : results_stk;
1183       
1184       for (j = 0; j < p->data.or.n; j++) {
1185         if (mpc_parse_run(i, p->data.or.xs[j], &results[j], e)) {
1186           MPC_SUCCESS(results[j].output;
1187             if (p->data.or.n > MPC_PARSE_STACK_MIN) { mpc_free(i, results); });
1188         } else {
1189           *e = mpc_err_merge(i, *e, results[j].error);
1190         } 
1191       }
1192       
1193       MPC_FAILURE(NULL;
1194         if (p->data.or.n > MPC_PARSE_STACK_MIN) { mpc_free(i, results); });
1195     
1196     case MPC_TYPE_AND:
1197       
1198       if (p->data.and.n == 0) { MPC_SUCCESS(NULL); }
1199       
1200       results = p->data.or.n > MPC_PARSE_STACK_MIN
1201         ? mpc_malloc(i, sizeof(mpc_result_t) * p->data.or.n)
1202         : results_stk;
1203       
1204       mpc_input_mark(i);
1205       for (j = 0; j < p->data.and.n; j++) {
1206         if (!mpc_parse_run(i, p->data.and.xs[j], &results[j], e)) {
1207           mpc_input_rewind(i);
1208           for (k = 0; k < j; k++) {
1209             mpc_parse_dtor(i, p->data.and.dxs[k], results[k].output);
1210           }
1211           MPC_FAILURE(results[j].error;
1212             if (p->data.or.n > MPC_PARSE_STACK_MIN) { mpc_free(i, results); });
1213         }
1214       }
1215       mpc_input_unmark(i); 
1216       MPC_SUCCESS(
1217         mpc_parse_fold(i, p->data.and.f, j, (mpc_val_t**)results);
1218         if (p->data.or.n > MPC_PARSE_STACK_MIN) { mpc_free(i, results); });
1219     
1220     /* End */
1221     
1222     default:
1223       
1224       MPC_FAILURE(mpc_err_fail(i, "Unknown Parser Type Id!"));
1225   }
1226   
1227   return 0;
1228   
1229 }

1959 /*

1960 ** So here is a cute bootstrapping.

1961 **

1962 ** I'm using the previously defined

1963 ** mpc constructs and functions to

1964 ** parse the user regex string and

1965 ** construct a parser from it.

1966 **

1967 ** As it turns out lots of the standard

1968 ** mpc functions look a lot like `fold`

1969 ** functions and so can be used indirectly

1970 ** by many of the parsing functions to build

1971 ** a parser directly - as we are parsing.

1972 **

1973 ** This is certainly something that

1974 ** would be less elegant/interesting 

1975 ** in a two-phase parser which first

1976 ** builds an AST and then traverses it

2096 static mpc_val_t *mpcf_re_range(mpc_val_t *x) {
2097   
2098   mpc_parser_t *out;
2099   size_t i, j;
2100   size_t start, end;
2101   const char *tmp = NULL;
2102   const char *s = x;
2103   int comp = s[0] == '^' ? 1 : 0;
2104   char *range = calloc(1,1);
2105   
2106   if (s[0] == '\0') { free(x); return mpc_fail("Invalid Regex Range Expression"); } 
2107   if (s[0] == '^' && 
2108       s[1] == '\0') { free(x); return mpc_fail("Invalid Regex Range Expression"); }
2109   
2110   for (i = comp; i < strlen(s); i++){
2111     
2112     /* Regex Range Escape */
2113     if (s[i] == '\\') {
2114       tmp = mpc_re_range_escape_char(s[i+1]);
2115       if (tmp != NULL) {
2116         range = realloc(range, strlen(range) + strlen(tmp) + 1);
2117         strcat(range, tmp);
2118       } else {
2119         range = realloc(range, strlen(range) + 1 + 1);
2120         range[strlen(range) + 1] = '\0';
2121         range[strlen(range) + 0] = s[i+1];      
2122       }
2123       i++;
2124     }
2125     
2126     /* Regex Range...Range */
2127     else if (s[i] == '-') {
2128       if (s[i+1] == '\0' || i == 0) {
2129           range = realloc(range, strlen(range) + strlen("-") + 1);
2130           strcat(range, "-");
2131       } else {
2132         start = s[i-1]+1;
2133         end = s[i+1]-1;
2134         for (j = start; j <= end; j++) {
2135           range = realloc(range, strlen(range) + 1 + 1 + 1);
2136           range[strlen(range) + 1] = '\0';
2137           range[strlen(range) + 0] = (char)j;
2138         }        
2139       }
2140     }
2141     
2142     /* Regex Range Normal */
2143     else {
2144       range = realloc(range, strlen(range) + 1 + 1);
2145       range[strlen(range) + 1] = '\0';
2146       range[strlen(range) + 0] = s[i];
2147     }
2148   
2149   }
2150   
2151   out = comp == 1 ? mpc_noneof(range) : mpc_oneof(range);
2152   
2153   free(x);
2154   free(range);
2155   
2156   return out;

2160   
2161   char *err_msg;
2162   mpc_parser_t *err_out;

2169   Base   = mpc_new("base");

2171   
2172   mpc_define(Regex, mpc_and(2, mpcf_re_or,
2173     Term, 
2174     mpc_maybe(mpc_and(2, mpcf_snd_free, mpc_char('|'), Regex, free)),
2175     (mpc_dtor_t)mpc_delete
2176   ));
2177   
2178   mpc_define(Term, mpc_many(mpcf_re_and, Factor));
2179   
2180   mpc_define(Factor, mpc_and(2, mpcf_re_repeat,
2181     Base,
2182     mpc_or(5,
2183       mpc_char('*'), mpc_char('+'), mpc_char('?'),
2184       mpc_brackets(mpc_int(), free),
2185       mpc_pass()),
2186     (mpc_dtor_t)mpc_delete
2187   ));
2188   
2189   mpc_define(Base, mpc_or(4,
2190     mpc_parens(Regex, (mpc_dtor_t)mpc_delete),
2191     mpc_squares(Range, (mpc_dtor_t)mpc_delete),
2192     mpc_apply(mpc_escape(), mpcf_re_escape),
2193     mpc_apply(mpc_noneof(")|"), mpcf_re_escape)
2194   ));
2195   
2196   mpc_define(Range, mpc_apply(
2197     mpc_many(mpcf_strfold, mpc_or(2, mpc_escape(), mpc_noneof("]"))),
2198     mpcf_re_range
2199   ));
2200   
2201   RegexEnclose = mpc_whole(mpc_predictive(Regex), (mpc_dtor_t)mpc_delete);
2202   
2203   mpc_optimise(RegexEnclose);
2204   mpc_optimise(Regex);
2205   mpc_optimise(Term);
2206   mpc_optimise(Factor);
2207   mpc_optimise(Base);
2208   mpc_optimise(Range);
2209   
2210   if(!mpc_parse("<mpc_re_compiler>", re, RegexEnclose, &r)) {
2211     err_msg = mpc_err_string(r.error);
2212     err_out = mpc_failf("Invalid Regex: %s", err_msg);
2213     mpc_err_delete(r.error);  
2214     free(err_msg);
2215     r.output = err_out;
2216   }
2217   
2218   mpc_cleanup(6, RegexEnclose, Regex, Term, Factor, Base, Range);
2219   
2220   mpc_optimise(r.output);
2221   
2222   return r.output;
2223   
2224 }

2900 mpc_ast_t *mpc_ast_traverse_next(mpc_ast_trav_t **trav) {
2901   mpc_ast_trav_t *n_trav, *to_free;
2902   mpc_ast_t *ret = NULL;
2903   int cchild;
2904 
2905   /* The end of traversal was reached */
2906   if(*trav == NULL) return NULL;
2907 
2908   switch((*trav)->order) {
2909     case mpc_ast_trav_order_pre:
2910       ret = (*trav)->curr_node;
2911 
2912       /* If there aren't any more children, go up */
2913       while(*trav != NULL &&
2914         (*trav)->curr_child >= (*trav)->curr_node->children_num)
2915       {
2916         to_free = *trav;
2917         *trav = (*trav)->parent;
2918         free(to_free);
2919       }
2920 
2921       /* If trav is NULL, the end was reached */
2922       if(*trav == NULL) {
2923         break;
2924       }
2925 
2926       /* Go to next child */
2927       n_trav = malloc(sizeof(mpc_ast_trav_t));
2928 
2929       cchild = (*trav)->curr_child;
2930       n_trav->curr_node = (*trav)->curr_node->children[cchild];
2931       n_trav->parent = *trav;
2932       n_trav->curr_child = 0;
2933       n_trav->order = (*trav)->order;
2934 
2935       (*trav)->curr_child++;
2936       *trav = n_trav;
2937 
2938       break;
2939 
2940     case mpc_ast_trav_order_post:
2941       ret = (*trav)->curr_node;
2942 
2943       /* Move up tree to the parent If the parent doesn't have any more nodes,

2944        * then this is the current node. If it does, move down to its left most

2945        * child. Also, free the previous traversal node */
2946       to_free = *trav;
2947       *trav = (*trav)->parent;
2948       free(to_free);
2949 
2950       if(*trav == NULL)
2951         break;
2952 
2953       /* Next child */
2954       (*trav)->curr_child++;
2955 
2956       /* If there aren't any more children, this is the next node */
2957       if((*trav)->curr_child >= (*trav)->curr_node->children_num) {
2958         break;
2959       }
2960 
2961       /* If there are still more children, find the leftmost child from this

2962        * node */
2963       while((*trav)->curr_node->children_num > 0) {
2964         n_trav = malloc(sizeof(mpc_ast_trav_t));
2965 
2966         cchild = (*trav)->curr_child;
2967         n_trav->curr_node = (*trav)->curr_node->children[cchild];
2968         n_trav->parent = *trav;
2969         n_trav->curr_child = 0;
2970         n_trav->order = (*trav)->order;
2971 
2972         *trav = n_trav;
2973       }
2974 
2975     default:
2976       /* Unreachable, but compiler complaints */
2977       break;
2978   }
2979

3123 **

3126 ** let all fold rules be automatically taken

3127 ** care of by existing functions.

3135 ** like regex but the main difference is that

3136 ** it is now whitespace insensitive and the

3137 ** base type takes literals of some form.

3140 /*

3141 **

3142 **  ### Grammar Grammar

3143 **

3144 **      <grammar> : (<term> "|" <grammar>) | <term>

3145 **     

3146 **      <term> : <factor>*

3147 **

3148 **      <factor> : <base>

3149 **               | <base> "*"

3150 **               | <base> "+"

3151 **               | <base> "?"

3152 **               | <base> "{" <digits> "}"

3153 **           

3154 **      <base> : "<" (<digits> | <ident>) ">"

3155 **             | <string_lit>

3156 **             | <char_lit>

3157 **             | <regex_lit>

3158 **             | "(" <grammar> ")"

3159 */

3162   va_list *va;
