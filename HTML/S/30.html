  79 static Yes_No can_move_horizon(Camera_Access access,
  80     Point_Type point) {
  81   Point_Type center = access->center;
  82 
  83   Point_Type from = access->start;
  84   Point_Type to = access->end;
  85 
  86   Map_Access map = access->map;

  90       && ((point.x + 1)< (map->end.x - (center.x - 1)))) {
  91     access->horizon = CAMERA_MOVE;
  92   }
  93   else if (from.x <= 0) {
  94     access->horizon = CAMERA_FIX;
  95 
  96     if (point.x < 0) {
  97       result = NO;
  98     }
  99   }
 100   else if (to.x >= map->end.x) {
 101     access->horizon = CAMERA_FIX;
 102 
 103     if (point.x >= map->end.x) {
 104       result = NO;
 105     }
 106   }
 107   else {
 108     access->horizon = CAMERA_MOVE;

 147 static bool key_process(Camera_Access access, Character_Pool_Access from_pool,
 148     Message_Box_Access box_access, SDL_Event *event) {
 149   Status_Access Player = access->player;
 150   Status_Access npc = NULL;
 151   Style_Access dead = access->dead;
 152 
 153   Point_Type point = {
 154     Player->base->Real_Position.x,
 155     Player->base->Real_Position.y
 156   };
 157 
 158   Point_Type vector = {0, 0};
 159 
 160   if ((event->key).keysym.sym == SDLK_q) {
 161     return false;
 162   }
 163 
 164   switch ((event->key).keysym.sym) {
 165     case SDLK_UP:
 166       point.y -= 1;
 167       vector.y = -1;
 168       break;
 169     case SDLK_DOWN:
 170       point.y += 1;
 171       vector.y = 1;
 172       break;
 173     case SDLK_LEFT:
 174       point.x -= 1;
 175       vector.x = -1;
 176       break;
 177     case SDLK_RIGHT:
 178       point.x += 1;
 179       vector.x = 1;
 180       break;
 181     default:
 182       break;
 183   }
 184 
 185   if (occupy_position_by_others(from_pool, point, &npc) == NO) {
 186     if ((vector.y != 0) && (can_move_vertical(access, point) == YES)) {
 187       switch (access->vertical) {
 188         case CAMERA_MOVE:
 189           access->start.y = access->start.y + vector.y;
 190           access->end.y = access->end.y + vector.y;
 191           break;
 192         default:
 193           break;
 194       }
 195 
 196       Player->base->Real_Position.x = Player->base->Real_Position.x + vector.x;
 197       Player->base->Real_Position.y = Player->base->Real_Position.y + vector.y;
 198     }
 199     else if ((vector.x != 0) && (can_move_horizon(access, point) == YES)) {
 200       switch (access->horizon) {
 201         case CAMERA_MOVE:
 202           access->start.x = access->start.x + vector.x;
 203           access->end.x = access->end.x + vector.x;
 204           break;
 205         default:
 206           break;
 207       }
 208 
 209       Player->base->Real_Position.x = Player->base->Real_Position.x + vector.x;
 210       Player->base->Real_Position.y = Player->base->Real_Position.y + vector.y;
 211     }
 212   }
 213   else {
 214     if (character.attack(Player, npc) == DEAD) {
 215       character.set_style(npc, dead);
 216 
 217       char *a = " 死亡";
 218       int64_t len = string.strlen_ascii(a);
 219       len += string.strlen_ascii(npc->base->name);
 220       String message = string_pool.malloc(tmp_pool, len);
 221       strcpy(message, npc->base->name);
 222       strcat(message, a);
 223       message_box.add(box_access, message);
 224       string_pool.reset(tmp_pool);
 225     }
 226     else {
 227       char *a = "攻擊 ";
 228       char *b = "，受到 1 點傷害";
 229       int64_t len = string.strlen_ascii(a) + string.strlen_ascii(b);
 230       len += string.strlen_ascii(npc->base->name);
 231       String message = string_pool.malloc(tmp_pool, len);
 232       strcpy(message, a);
 233       strcat(message, npc->base->name);
 234       strcat(message, b);
 235       message_box.add(box_access, message);
 236       string_pool.reset(tmp_pool);
 237     }
 238   }
 239   character_pool.reset_graph_position(from_pool,
 240       access->start.x, access->start.y,
 241       access->max_x, access->max_y);
 242   return true;
 243 }
