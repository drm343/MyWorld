  99 void setup_npc(char *file_path) {
 100   mpc_result_t r;
 101 
 102   mpc_parser_t* Number         = mpc_new("number");
 103   mpc_parser_t* Key            = mpc_new("key");
 104   mpc_parser_t* Json_String    = mpc_new("json_string");
 105   mpc_parser_t* Json_Boolean   = mpc_new("json_boolean");
 106   mpc_parser_t* Json_Value     = mpc_new("json_value");
 107   mpc_parser_t* KV_Item        = mpc_new("kv_item");
 108   mpc_parser_t* Json_Array     = mpc_new("json_array");
 109   mpc_parser_t* Json_Object    = mpc_new("json_obj");
 110   mpc_parser_t* JSON           = mpc_new("json");
 111 
 112   mpca_lang(MPCA_LANG_PREDICTIVE,
 113       " number         : /[0-9]+/                              ; \n"
 114       " key            : /[^'\"''{''}'':'';'' '',']*/          ; \n"
 115       " json_string    : '\"' <key> '\"'                       ; \n"
 116       " json_boolean   : \"true\" | \"false\"                  ; \n"
 117       " kv_item        : <json_string> ':'                         "
 118       "                   (<json_value> | <json_array>)        ; \n"
 119       " json_array     : '[' (<json_value> ','?)* ']'          ; \n"
 120       " json_obj       : '{' (<kv_item> ','?)* '}'             ; \n"
 121       " json_value     : <json_obj> | <json_array>                 "
 122       "                | <json_boolean>                            "
 123       "                | <json_string> | <number>              ; \n"
 124       " json           : /^/ <json_value> /$/                  ; \n",
 125       Number, Key,
 126       Json_String, Json_Boolean,
 127       KV_Item,
 128       Json_Array, Json_Object,
 129       Json_Value,
 130       JSON, NULL);
 131 
 132   if (mpc_parse_contents(file_path, JSON, &r)) {
 133     //mpc_ast_print(r.output);
 134     eval_set_npc(r.output);
 135     mpc_ast_delete(r.output);
 136   } else {
 137     mpc_err_print(r.error);
 138     mpc_err_delete(r.error);
 139   }
 140 
 141   mpc_cleanup(9,
 142       Number, Key,
 143       Json_String, Json_Boolean,
 144       KV_Item,
 145       Json_Array, Json_Object,
 146       Json_Value,
